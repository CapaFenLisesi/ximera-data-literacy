\begin{exercise}
Given that $s(k)=-k^2-4 k+2$, evaluate $s(4.5)$. Express your answer in decimal notation.
\begin{solution}
\begin{hint}
$s(4.5)=-(4.5)^2-4 (4.5)+2$.
\end{hint}
\begin{hint}
$s(4.5)=-36.25$.
\end{hint}
The value of the function $s(k)=-k^2-4 k+2$, evaluated at $k=4.5$, is \answer{-36.25}.
\end{solution}
\end{exercise}
